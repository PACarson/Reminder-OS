function createTriggers() {
  // 先删掉所有已存在的 checkReminders 触发器，避免重复跑这个函数会叠加
  // 出两个同名 trigger（这点是我加的防御性处理，你原来的版本不一定有）
  var triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(function (t) {
    if (t.getHandlerFunction() === 'checkReminders') {
      ScriptApp.deleteTrigger(t);
    }
  });

  ScriptApp.newTrigger('checkReminders')
    .timeBased()
    .everyHours(1)
    .create();

  Logger.log('[Setup] checkReminders 触发器已建立（每小时）');
  return { ok: true };
}

function runDiagnostics() {
  var result = { ok: true, checks: {} };

  // 1. 能否读到共享 Spreadsheet 的 Tasks 表
  try {
    var tasks = QueryEngine.getPendingTasks();
    result.checks.tasks_sheet_readable = true;
    result.checks.pending_task_count = tasks.length;
  } catch (e) {
    result.ok = false;
    result.checks.tasks_sheet_readable = false;
    result.checks.tasks_sheet_error = e.message;
  }

  // 2. 能否发出一条测试 Telegram 消息
  try {
    var chatId = SecureConfig.getKey('TELEGRAM_CHAT_ID');
    var sendResult = Output.sendMessage(chatId, '✅ Reminder OS runDiagnostics() 测试消息');
    result.checks.telegram_send = !!(sendResult && sendResult.ok);
  } catch (e) {
    result.ok = false;
    result.checks.telegram_send = false;
    result.checks.telegram_error = e.message;
  }

  Logger.log('[Setup] runDiagnostics 结果: ' + JSON.stringify(result));
  return result;
}
