/**
 * ConnectorResponse.gs
 * Personal AI Core — Connector Layer / 统一响应 DTO 工厂
 *
 * 【新增文件】Connector Layer V1 落地的一部分，完整背景/架构见本目录
 * README.gs，决策记录见 00_Project_Constitution.gs 二/平台架构 D5(3)。
 *
 * 职责（且仅有这一个职责——本文件不做校验、不做日志、不知道任何具体
 * Domain OS 是什么）：
 *   把"这次调用的结果"统一包成一个形状固定的对象，任何 Connector 方法
 *   （create/update/delete/get/list/search/execute/health/capabilities）
 *   最终返回给调用方的都必须是这个形状——不允许任何 Connector 直接
 *   return 裸的 task/reminder 对象或裸的 Error。
 *
 * 响应形状（字段名固定，来自需求方规格）：
 *   {
 *     success:   boolean,
 *     data:      any | null,
 *     error:     { category, code, message, details } | null,
 *     message:   string,
 *     metadata:  object,          // 调用方上下文（connector/method/action/chatId等），
 *                                 // 不放业务数据本体（业务数据只放 data）
 *     timestamp: string,         // ISO 8601，响应构造的时刻
 *     duration:  number,         // 毫秒，调用耗时（由调用方在 ConnectorTypes.gs
 *                                 // 的 ConnectorSupport.handle 里измерен后传入）
 *     traceId:   string          // 本次调用的唯一追踪号，串联日志用
 *   }
 *
 * 本文件不依赖任何其他 Connector Layer 文件（ConnectorTypes.gs 反过来依赖
 * 本文件——ConnectorSupport.handle 内部调用 ConnectorResponse.success/
 * failure 来产出最终返回值）。也不依赖任何 Domain OS / Shared Service。
 *
 * ── Connector Contract（借用 Productivity OS 00_Project_Constitution.gs
 *    零之三 Engine Contract Standard 的字段格式，适配成 Connector 版本；
 *    Personal AI Core 自己的 00_Project_Constitution.gs 目前没有这一级的
 *    强制标准，这里主动采用是因为这个格式本身确实好用，沿用比自创一套
 *    更省心，见 README.gs「设计取舍」一节）──
 *   Responsibilities      : 统一响应对象的形状与构造
 *   Owns                  : success()/failure() 两个工厂函数的字段拼装规则
 *   Reads                 : 调用方传入的 data/error/message/meta 参数
 *   Writes                : none
 *   Public API            : ConnectorResponse.success(data, opts),
 *                           ConnectorResponse.failure(errorInfo, opts)
 *   Dependencies          : 无（纯函数模块）
 *   Forbidden Dependencies: 不得依赖 ConnectorTypes.gs / 任何 Connector /
 *                           任何 Domain OS / Sheet / Events
 *   Pure Function         : YES
 *   Replay Events         : NO
 *   Projection            : NO
 *   Thread Safety         : 不需要（无共享可变状态）
 *   Side Effects          : NO
 *   Notes                 : traceId/timestamp/duration 由调用方（通常是
 *                           ConnectorTypes.gs 的 ConnectorSupport.handle）
 *                           算好后通过 opts 传入；本文件本身不生成，保持
 *                           纯函数、方便单测（不用 mock Date.now()）。
 */

var ConnectorResponse = (function () {

  /**
   * @param {any} data              业务数据本体，成功时返回
   * @param {object} [opts]
   * @param {string} [opts.message]   人类可读的一句话摘要
   * @param {object} [opts.metadata]  非业务性的调用上下文
   * @param {string} [opts.timestamp] ISO 8601；不传则用当前时间
   * @param {number} [opts.duration]  毫秒；不传则为 0
   * @param {string} [opts.traceId]   不传则为空字符串（正常路径不应该不传，
   *                                  见 ConnectorTypes.gs 的 ConnectorSupport.handle）
   * @returns {object} 标准响应对象
   */
  function success(data, opts) {
    opts = opts || {};
    return {
      success:   true,
      data:      (data === undefined) ? null : data,
      error:     null,
      message:   opts.message || 'OK',
      metadata:  opts.metadata || {},
      timestamp: opts.timestamp || new Date().toISOString(),
      duration:  (typeof opts.duration === 'number') ? opts.duration : 0,
      traceId:   opts.traceId || ''
    };
  }

  /**
   * @param {object} errorInfo
   * @param {string} errorInfo.category  ConnectorErrorCategory 四选一
   *                                     （VALIDATION_ERROR/BUSINESS_ERROR/
   *                                     SYSTEM_ERROR/UNKNOWN_ERROR，见
   *                                     ConnectorTypes.gs）
   * @param {string} errorInfo.code      具体错误码，比如 'TASK_NOT_FOUND'
   * @param {string} errorInfo.message   人类可读的错误说明
   * @param {any}    [errorInfo.details] 可选，结构化补充信息（不放敏感数据）
   * @param {object} [opts]              同 success() 的 opts
   * @returns {object} 标准响应对象（success:false）
   */
  function failure(errorInfo, opts) {
    opts = opts || {};
    errorInfo = errorInfo || {};
    return {
      success:   false,
      data:      null,
      error: {
        category: errorInfo.category || 'UNKNOWN_ERROR',
        code:     errorInfo.code     || 'UNKNOWN_ERROR',
        message:  errorInfo.message  || 'An unknown error occurred.',
        details:  (errorInfo.details === undefined) ? null : errorInfo.details
      },
      message:   opts.message || errorInfo.message || 'Request failed.',
      metadata:  opts.metadata || {},
      timestamp: opts.timestamp || new Date().toISOString(),
      duration:  (typeof opts.duration === 'number') ? opts.duration : 0,
      traceId:   opts.traceId || ''
    };
  }

  /**
   * 快速判断一个对象是不是本工厂产出的标准响应形状——用于 Registry 的
   * healthCheckAll()/capabilitiesAll() 之类"聚合别人返回值"的场景做防御性
   * 检查（比如某个 connector 的方法万一没走 ConnectorSupport.handle，
   * 直接 return 了裸对象）。只做形状检查，不做语义校验。
   * @param {any} obj
   * @returns {boolean}
   */
  function isConnectorResponse(obj) {
    return !!obj &&
      typeof obj === 'object' &&
      typeof obj.success === 'boolean' &&
      typeof obj.timestamp === 'string' &&
      typeof obj.traceId === 'string' &&
      ('data' in obj) &&
      ('error' in obj);
  }

  return {
    success:             success,
    failure:             failure,
    isConnectorResponse: isConnectorResponse
  };
})();
