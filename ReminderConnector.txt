/**
 * ReminderConnector.gs
 * Personal AI Core — Connector Layer / Reminder OS 的唯一公开通信边界
 *
 * 【新增文件】Connector Layer V1 落地的一部分，完整背景/架构见本目录
 * README.gs，决策记录见 00_Project_Constitution.gs 二/平台架构 D5(3)。
 *
 * ⚠️⚠️ 写这个文件之前必须理解的现实（不是设计选择，是 Reminder OS 当前
 * 真实的物理形态——见其 00_Project_Constitution.gs P2/P3）：
 *
 *   Reminder OS 是完全独立运作的 Apps Script 项目：自己的每小时时间触发器
 *   （checkReminders），不接受任何项目把它当 Library 调用，也不接受
 *   webhook，"不被任何项目调用，也不调用任何项目"。它跟 Personal AI Core /
 *   Productivity OS 唯一的联系是——三个项目的 Script Properties 里
 *   SPREADSHEET_ID 被人工配置成同一个值，读写同一张共享 Google Sheet：
 *     - 读全部字段：Tasks 表（Productivity OS 的地盘）
 *     - 写两个字段：Tasks 表的 reminder_count / last_reminder_at
 *     - 只追加：Events 表的 REMINDER_SENT 事件
 *   它没有"提醒"这个独立实体/独立表——V1 的"提醒"就是"每小时扫一遍
 *   Tasks 表里状态为 PENDING 的任务，用 due_date+priority 算一下该不该
 *   发"，算完直接发 Telegram，不经过任何人审批，也不能被外部单独触发/
 *   创建/修改/取消/延后。
 *
 *   结论：CreateReminder/UpdateReminder/DeleteReminder/DismissReminder/
 *   SnoozeReminder/TriggerReminder 六个操作，在 Reminder OS 当前版本下
 *   【没有对应的真实后端能力可以路由】——不是本文件没写，是根本没有
 *   "Reminder OS 自己暴露的入口"可以调用（Connector 的职责是"路由请求
 *   进 Domain OS"，不是在找不到入口时凭空发明一套逻辑去模拟出"看起来
 *   支持"的假象——那是重复实现业务逻辑，正是需求方"Do NOT duplicate
 *   business logic"明确禁止的事，也违反 Reminder OS 自己
 *   00_Project_Constitution.gs 的"提醒逻辑只允许存在一套实现"）。
 *   这六个操作会诚实地返回 BUSINESS_ERROR（能力不存在），并在
 *   capabilities() 里可以提前查到，不需要先调用一次才知道。
 *
 *   真正可以做、且这里是真实实现（不是占位）的，是【读】：
 *   UpcomingReminders / TodayReminders / Statistics / get / list / search——
 *   全部通过 Personal AI Core 自己已有的 Shared Services（12_QueryEngine.gs
 *   的全局 QueryEngine、02_EventBus.gs 的全局 EventBus、05_SheetUtils.gs
 *   的 isOverdue_/parseDueDate_）读同一张共享 Sheet，不直接操作 Sheet
 *   （复用现成读取层，不重新实现一遍"扫表"逻辑），也完全不调用
 *   ProductivityOS Library（"每个 Domain OS 恰好一个 Connector"——即使
 *   读的是同一批 task_id，Reminder 相关的请求也只走这个文件，Task
 *   本身的增删改查请求走 ProductivityConnector.gs）。
 *
 * ── Connector Contract ───────────────────────────────────────────────────
 *   Responsibilities      : 提醒相关只读查询（今日/即将到来/统计）的唯一
 *                           入口；对没有真实后端能力的写操作给出诚实、
 *                           标准化的能力缺口说明
 *   Owns                  : 无业务判断——本文件不复刻 Reminder OS 的
 *                           _shouldRemind 决策逻辑（那依赖
 *                           REMINDER_ADVANCE_HOURS/REMINDER_INTERVAL_HOURS，
 *                           只存在于 Reminder OS 项目自己的文件里）。
 *                           is_overdue 字段只是"到期时间是否已过"的事实
 *                           陈述，不等于"现在会不会被提醒"
 *   Calls Into            : 无 Library、无 webhook——见文件头说明。只读
 *                           Personal AI Core 本项目自己的全局
 *                           QueryEngine（12_QueryEngine.gs）/
 *                           EventBus（02_EventBus.gs）/
 *                           isOverdue_、parseDueDate_（05_SheetUtils.gs）
 *   Does NOT Call          : ProductivityOS Library 的任何函数（哪怕是
 *                           只读查询，也要用 Reminder OS 自己能看到的
 *                           同一份共享 Sheet 数据——Core 本地的
 *                           QueryEngine/EventBus 本来就是读这同一张表，
 *                           不需要绕去 ProductivityOS Library）
 *   Public API             : create, update, delete, get, list, search,
 *                            execute, health, capabilities（Standard
 *                            Connector Interface，见 README.gs）
 *   Dependencies           : ConnectorTypes.gs、ConnectorResponse.gs（间接）、
 *                            Personal AI Core 全局 QueryEngine / EventBus /
 *                            isOverdue_ / parseDueDate_
 *   Forbidden Dependencies : ProductivityOS Library；Reminder OS 的任何
 *                            Library/webhook（不存在，也不应该假装存在）
 *   Thread Safety          : 纯读取，不需要加锁
 *   Side Effects           : NO
 *   Notes                  : CompleteReminder/DismissReminder/SnoozeReminder
 *                            为什么不支持、该去哪里做，见 capabilities()
 *                            和对应 case 分支的详细说明。
 * ─────────────────────────────────────────────────────────────────────────
 */

var ReminderConnector = (function () {

  var CONNECTOR_NAME = ConnectorNames.REMINDER;
  var DEFAULT_UPCOMING_DAYS = 7;

  // ============ 内部：共享 Sheet 读取（全部经由 Core 本地既有 Shared Service） ============

  function _ensureSharedSheetReachable_() {
    try {
      getSheet_('Tasks'); // 05_SheetUtils.gs，Core 本地既有，读同一张共享 Sheet
    } catch (e) {
      throw ConnectorErrors.system(
        'SHARED_SHEET_UNAVAILABLE',
        'Could not reach the shared Tasks sheet: ' + e.message,
        null
      );
    }
  }

  /**
   * 把一个 task 对象投影成"提醒视角"的字段子集——只陈述事实（Reminder OS
   * 实际读写过的字段 + due_date 是否已过期这个纯日期判断），不重新计算
   * "该不该提醒"（那是 Reminder OS 自己的 _shouldRemind，依赖只存在于
   * 它项目里的 REMINDER_ADVANCE_HOURS/REMINDER_INTERVAL_HOURS 配置）。
   */
  function _reminderView_(task) {
    return {
      task_id:          task.task_id,
      title:            task.title,
      chat_id:          task.chat_id,
      status:           task.status,
      priority:         task.priority,
      due_date:         task.due_date || '',
      is_overdue:       isOverdue_(task.due_date), // 05_SheetUtils.gs，纯日期比较，不是"该不该提醒"的决策
      reminder_count:   Number(task.reminder_count) || 0,
      last_reminder_at: task.last_reminder_at || null
    };
  }

  function _matchesChatId_(task, chatId) {
    return !chatId || String(task.chat_id) === String(chatId);
  }

  /**
   * 距 due_date 还有多少天（本地时区午夜对齐），无法解析（含里程类
   * '40000km'）时返回 null。只用来做 Today/Upcoming 的日期分桶——纯日期
   * 运算，复用 05_SheetUtils.gs 的 parseDueDate_，不是新发明的解析逻辑。
   */
  function _daysUntilDue_(dueDateRaw) {
    if (!dueDateRaw) return null;
    var raw = String(dueDateRaw).trim();
    if (/km$/i.test(raw)) return null;

    var due = parseDueDate_(raw); // 05_SheetUtils.gs
    if (!due || isNaN(due.getTime())) return null;

    var tz = Session.getScriptTimeZone();
    var todayKey = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd');
    var dueKey   = Utilities.formatDate(due, tz, 'yyyy-MM-dd');
    var todayMidnight = parseDueDate_(todayKey);
    var dueMidnight   = parseDueDate_(dueKey);

    return Math.round((dueMidnight.getTime() - todayMidnight.getTime()) / 86400000);
  }

  // ============ create() / update() / delete() → 无真实后端能力 ============

  var WRITE_UNSUPPORTED_MESSAGE =
    'Reminder OS v1.0 has no writable "reminder" entity — reminders are computed hourly ' +
    'from Productivity OS\'s Tasks (due_date + priority), not stored/created independently. ' +
    'See Reminder OS 00_Project_Constitution.gs P1–P3 and 00_ADR_003 (proposed V2 scope, not yet built).';

  function create(payload, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'create', action: 'CreateReminder', chatId: context && context.chatId },
      function () {
        throw ConnectorErrors.business('REMINDER_WRITE_UNSUPPORTED', WRITE_UNSUPPORTED_MESSAGE);
      }
    );
  }

  function update(id, changes, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'update', action: 'UpdateReminder', chatId: context && context.chatId },
      function () {
        throw ConnectorErrors.business('REMINDER_WRITE_UNSUPPORTED', WRITE_UNSUPPORTED_MESSAGE);
      }
    );
  }

  function del(id, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'delete', action: 'DeleteReminder', chatId: context && context.chatId },
      function () {
        throw ConnectorErrors.business('REMINDER_WRITE_UNSUPPORTED', WRITE_UNSUPPORTED_MESSAGE);
      }
    );
  }

  // ============ get() → 按 task_id 读"提醒视角"的状态 ============

  /**
   * @param {string} id       task_id（V1 里提醒没有独立 id，借用它所属
   *                          任务的 task_id）
   * @param {object} [context] { chatId? }（传入时做跨用户隔离校验）
   * @returns {object} ConnectorResponse；data 是 _reminderView_ 形状
   */
  function get(id, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'get', action: 'GetReminder', chatId: context && context.chatId },
      function () {
        _ensureSharedSheetReachable_();
        if (!ConnectorValidation.isNonEmptyString(id)) {
          throw ConnectorErrors.validation('TASK_ID_REQUIRED', 'id (task_id, non-empty string) is required.');
        }

        var task = QueryEngine.getTaskById(id); // 12_QueryEngine.gs，Core 本地既有
        var chatId = context && context.chatId;
        if (!task || !_matchesChatId_(task, chatId)) {
          throw ConnectorErrors.business('TASK_NOT_FOUND', 'No task found with id "' + id + '".');
        }

        return { data: _reminderView_(task) };
      }
    );
  }

  // ============ list() → 所有被 Reminder OS 监控中的任务（即 PENDING 任务） ============

  /**
   * @param {object} [filters]  { onlyReminded?: boolean }——true 时只保留
   *                             reminder_count>0 的任务（已经被提醒过至少
   *                             一次）
   * @param {object} [context]  { chatId? }
   * @returns {object} ConnectorResponse；data 是 _reminderView_[] 数组
   */
  function list(filters, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'list', action: 'ListReminders', chatId: context && context.chatId },
      function () {
        _ensureSharedSheetReachable_();
        if (filters !== undefined && filters !== null && !ConnectorValidation.isPlainObject(filters)) {
          throw ConnectorErrors.validation('INVALID_FILTERS', 'filters must be a plain object, if provided.');
        }

        var tasks = QueryEngine.getPendingTasks(context && context.chatId);
        if (filters && filters.onlyReminded) {
          tasks = tasks.filter(function (t) { return (Number(t.reminder_count) || 0) > 0; });
        }

        var views = tasks.map(_reminderView_);
        return { data: views, metaExtra: { count: views.length } };
      }
    );
  }

  // ============ search() → 按标题关键字过滤"被监控中的任务" ============

  /**
   * @param {string|object} query  字符串或 {text: string}——只支持标题
   *                                子串匹配（大小写不敏感）。Reminder OS
   *                                没有自己的搜索引擎，这里不引入
   *                                Productivity OS 的 23_SearchEngine.gs
   *                                （会违反"只跟 Reminder OS 通信"的边界），
   *                                保持故意简单
   * @param {object} [context]     { chatId? }
   * @returns {object} ConnectorResponse
   */
  function search(query, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'search', action: 'SearchReminders', chatId: context && context.chatId },
      function () {
        _ensureSharedSheetReachable_();
        if (!ConnectorValidation.isStringOrPlainObject(query)) {
          throw ConnectorErrors.validation('INVALID_QUERY', 'query must be a string or a plain object like {text: "..."}.');
        }
        var text = (typeof query === 'string') ? query : (query && query.text) || '';
        var textLower = String(text).trim().toLowerCase();

        var tasks = QueryEngine.getPendingTasks(context && context.chatId);
        if (textLower) {
          tasks = tasks.filter(function (t) {
            return String(t.title || '').toLowerCase().indexOf(textLower) !== -1;
          });
        }

        var views = tasks.map(_reminderView_);
        return { data: views, metaExtra: { count: views.length } };
      }
    );
  }

  // ============ execute() —— 具名 action 的通用分发入口 ============

  function execute(action, params, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'execute', action: action, chatId: context && context.chatId },
      function () {
        params = params || {};
        if (!ConnectorValidation.isNonEmptyString(action)) {
          throw ConnectorErrors.validation('ACTION_REQUIRED', 'action (non-empty string) is required.');
        }

        switch (action) {

          case 'TodayReminders': {
            _ensureSharedSheetReachable_();
            var todayTasks = QueryEngine.getPendingTasks(context && context.chatId).filter(function (t) {
              return _daysUntilDue_(t.due_date) === 0;
            });
            return { data: todayTasks.map(_reminderView_), metaExtra: { count: todayTasks.length } };
          }

          case 'UpcomingReminders': {
            _ensureSharedSheetReachable_();
            var horizonDays = (typeof params.days === 'number' && params.days > 0) ? params.days : DEFAULT_UPCOMING_DAYS;
            var upcoming = QueryEngine.getPendingTasks(context && context.chatId).filter(function (t) {
              var d = _daysUntilDue_(t.due_date);
              return d !== null && d >= 0 && d <= horizonDays;
            });
            return { data: upcoming.map(_reminderView_), metaExtra: { count: upcoming.length, horizonDays: horizonDays } };
          }

          case 'Statistics': {
            _ensureSharedSheetReachable_();
            var pending = QueryEngine.getPendingTasks(context && context.chatId);
            var remindedCount = 0, totalReminders = 0, overdueCount = 0;
            pending.forEach(function (t) {
              var count = Number(t.reminder_count) || 0;
              if (count > 0) remindedCount++;
              totalReminders += count;
              if (isOverdue_(t.due_date)) overdueCount++;
            });

            var sentEvents = EventBus.getEventsByType('REMINDER_SENT'); // 02_EventBus.gs，Core 本地既有
            var chatId = context && context.chatId;
            if (chatId) {
              sentEvents = sentEvents.filter(function (e) { return String(e.chat_id) === String(chatId); });
            }
            var lastSentAt = sentEvents.length > 0
              ? sentEvents.reduce(function (max, e) { return e.timestamp > max ? e.timestamp : max; }, sentEvents[0].timestamp)
              : null;

            return {
              data: {
                monitored_pending_tasks: pending.length,
                tasks_reminded_at_least_once: remindedCount,
                total_reminders_sent_to_open_tasks: totalReminders,
                overdue_pending_tasks: overdueCount,
                reminder_sent_events_alltime: sentEvents.length,
                last_reminder_sent_at: lastSentAt
              }
            };
          }

          case 'CompleteReminder':
            throw ConnectorErrors.business(
              'REMINDER_ACTION_UNSUPPORTED',
              'Reminders have no identity separate from their task. To mark this resolved, complete the underlying task instead: ' +
              'ConnectorRegistry.get("Productivity").execute("CompleteTask", {id: <task_id>}, context) — the existing task_done: ' +
              'Telegram callback in 04_Main.gs already does exactly this; this connector intentionally does not duplicate that path.'
            );

          case 'DismissReminder':
            throw ConnectorErrors.business(
              'REMINDER_ACTION_UNSUPPORTED',
              'Reminder OS v1.0 has no "dismiss" concept — the only lifecycle is the hourly checkReminders() sweep over Tasks; ' +
              'there is nothing to mark dismissed independently of the task itself.'
            );

          case 'SnoozeReminder':
            throw ConnectorErrors.business(
              'REMINDER_ACTION_UNSUPPORTED',
              'No real snooze state exists anywhere in the system today. Reminder OS sends a "⏰ Snooze 1h" button, but Personal AI ' +
              'Core\'s existing task_snooze: callback handler (04_Main.gs) only replies with an acknowledgement message and does not ' +
              'change any stored state — the task will simply be re-evaluated on the next hourly sweep like any other pending task.'
            );

          case 'TriggerReminder':
            throw ConnectorErrors.business(
              'REMINDER_ACTION_UNSUPPORTED',
              'Reminder OS only exposes checkReminders() as a trigger-bound entry point for its own hourly schedule over ALL pending ' +
              'tasks — it is not designed for, and does not accept, ad hoc invocation for a single reminder from another project ' +
              '(see Reminder OS 00_Project_Constitution.gs P2: not callable by any other project).'
            );

          default:
            throw ConnectorErrors.validation(
              'UNKNOWN_ACTION',
              'Unknown action "' + action + '". See capabilities() for the full supported action list.',
              { action: action }
            );
        }
      }
    );
  }

  // ============ health() ============

  function health(context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'health', chatId: context && context.chatId },
      function () {
        try {
          var pending = QueryEngine.getPendingTasks(context && context.chatId);
          var sentEvents = EventBus.getEventsByType('REMINDER_SENT');
          var lastSentAt = sentEvents.length > 0
            ? sentEvents.reduce(function (max, e) { return e.timestamp > max ? e.timestamp : max; }, sentEvents[0].timestamp)
            : null;

          return {
            data: {
              status: 'UP',
              integrationMode: 'shared-spreadsheet (read-only; no library, no webhook)',
              monitoredPendingTasks: pending.length,
              lastReminderSentAt: lastSentAt,
              note: 'This reflects reachability of the shared Tasks/Events sheet only — it is not a verdict on whether Reminder OS\'s own hourly trigger is currently alive, since that project exposes no status endpoint of its own.'
            }
          };
        } catch (e) {
          return { data: { status: 'DOWN', reason: 'Could not read the shared Tasks/Events sheet: ' + e.message } };
        }
      }
    );
  }

  // ============ capabilities() ============

  function capabilities() {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'capabilities' },
      function () {
        return {
          data: {
            connector: CONNECTOR_NAME,
            connectorVersion: '1.0.0',
            domainOS: 'Reminder OS v1.0',
            integrationMode: 'shared-spreadsheet, read-only (no Library, no webhook — see Reminder OS 00_Project_Constitution.gs P2)',
            standardMethods: ['create', 'update', 'delete', 'get', 'list', 'search', 'execute', 'health', 'capabilities'],
            actions: {
              CreateReminder:    { supported: false, via: 'create',  reason: WRITE_UNSUPPORTED_MESSAGE },
              UpdateReminder:    { supported: false, via: 'update',  reason: WRITE_UNSUPPORTED_MESSAGE },
              DeleteReminder:    { supported: false, via: 'delete',  reason: WRITE_UNSUPPORTED_MESSAGE },
              CompleteReminder:  { supported: false, via: 'execute', reason: 'See execute("CompleteReminder") error message — use ProductivityConnector.execute("CompleteTask") instead.' },
              DismissReminder:   { supported: false, via: 'execute', reason: 'No "dismiss" concept exists in Reminder OS v1.0.' },
              SnoozeReminder:    { supported: false, via: 'execute', reason: 'No real snooze state exists anywhere in the system today (Telegram button is acknowledgement-only).' },
              TriggerReminder:   { supported: false, via: 'execute', reason: 'checkReminders() is a trigger-bound entry point, not designed for ad hoc external/per-item invocation.' },
              GetReminder:       { supported: true,  via: 'get' },
              ListReminders:     { supported: true,  via: 'list' },
              SearchReminders:   { supported: true,  via: 'search' },
              TodayReminders:    { supported: true,  via: 'execute' },
              UpcomingReminders: { supported: true,  via: 'execute' },
              Statistics:        { supported: true,  via: 'execute' }
            }
          }
        };
      }
    );
  }

  return {
    create:       create,
    update:       update,
    delete:       del,
    get:          get,
    list:         list,
    search:       search,
    execute:      execute,
    health:       health,
    capabilities: capabilities
  };
})();
