/**
 * ProductivityConnector.gs
 * Personal AI Core — Connector Layer / Productivity OS 的唯一公开通信边界
 *
 * 【新增文件】Connector Layer V1 落地的一部分，完整背景/架构见本目录
 * README.gs，决策记录见 00_Project_Constitution.gs 二/平台架构 D5(3)。
 *
 * ── Connector Contract ───────────────────────────────────────────────────
 *   Responsibilities      : Task 生命周期操作（create/update/cancel/complete）
 *                           与查询（get/list/search/今日/优先级/统计等）的
 *                           唯一入口，负责校验输入、转换 DTO、把请求路由进
 *                           Productivity OS Library、把结果包成标准
 *                           ConnectorResponse
 *   Owns                  : 无业务判断——本文件不决定"这个任务该不该建/
 *                           该不该完成"，那是 Productivity OS 自己的事
 *   Calls Into            : ProductivityOS（Library，Identifier 固定为
 *                           "ProductivityOS"，见项目 README.md 第2步）
 *                             - ProductivityOS.createTask / .updateTask /
 *                               .cancelTask / .completeTask（向后兼容全局
 *                               wrapper，见 Productivity OS 20_TaskEngine.gs）
 *                             - ProductivityOS.TaskQueryEngine.*（centralized
 *                               query API，见 Productivity OS
 *                               12_TaskQueryEngine.gs）
 *   Does NOT Call         : Reminder OS 的任何东西（"每个 Domain OS 恰好
 *                           一个 Connector"——提醒相关请求一律走
 *                           ReminderConnector.gs，即使两者都在读同一批
 *                           task_id）
 *   Public API             : create, update, delete, get, list, search,
 *                            execute, health, capabilities（Standard
 *                            Connector Interface，见 README.gs）
 *   Dependencies           : ConnectorTypes.gs（ConnectorSupport.handle /
 *                            ConnectorErrors / ConnectorValidation /
 *                            ConnectorNames）、ConnectorResponse.gs（间接，
 *                            通过 ConnectorSupport.handle）
 *   Forbidden Dependencies : 不得直接操作 Sheet/Events（Tasks/Events表的
 *                            读写权威完全在 Productivity OS 项目内部，见
 *                            其 00_File_Map.gs"架构铁律"第2/3条）；不得
 *                            调用 Reminder OS 的任何东西
 *   Thread Safety          : 本文件不加锁——create 路径的并发安全由
 *                            Productivity OS 自己的 09_IdempotencyManager
 *                            （ScriptLock）保证，Connector 只是转发调用，
 *                            不重复实现锁
 *   Side Effects           : create/update/delete/execute(CompleteTask)
 *                            会让 Productivity OS 发布 Events（间接副作用，
 *                            不是本文件自己写的）
 *   Notes                  : DeleteTask 与 ReopenTask 两个操作在当前
 *                            Productivity OS 版本下的真实语义见下方
 *                            capabilities() 和对应方法内的注释——不是漏做，
 *                            是这两个操作在事件溯源架构下本来就没有对应的
 *                            真实能力（硬删除违反"Events 不可变"；重新打开
 *                            需要改 status，但 status 不在 updateTask 的
 *                            UPDATABLE_FIELDS 白名单内，见 Productivity OS
 *                            20_TaskEngine.gs）。
 * ─────────────────────────────────────────────────────────────────────────
 */

var ProductivityConnector = (function () {

  var CONNECTOR_NAME = ConnectorNames.PRODUCTIVITY;

  // Productivity OS Task 对象里，创建时允许调用方指定的可选字段
  // （照抄 Productivity OS 20_TaskEngine.gs 的 createTaskDirect_ 会读取的
  // 字段——不是重新定义一份白名单，只是"哪些字段值得从 payload 里挑出来
  // 转发"，Productivity OS 自己该拒绝的非法值（比如不认识的 category）
  // 仍然由它自己判断，这里不重复判断）
  var CREATE_META_FIELDS = ['due_date', 'priority', 'category', 'recurring',
                             'context', 'budget', 'notes', 'description', 'tags'];

  // ============ 内部：前置检查 / 小工具 ============

  function _ensureLibraryLoaded_() {
    if (typeof ProductivityOS === 'undefined') {
      throw ConnectorErrors.system(
        'PRODUCTIVITY_OS_UNAVAILABLE',
        'ProductivityOS library is not loaded in this Apps Script project.',
        { hint: 'Editor → Libraries → Identifier must be exactly "ProductivityOS" (see project README.md step 2).' }
      );
    }
  }

  function _requireChatId_(context) {
    var chatId = context && context.chatId;
    if (!ConnectorValidation.isNonEmptyString(chatId)) {
      throw ConnectorErrors.validation('CHAT_ID_REQUIRED', 'context.chatId (non-empty string) is required for this operation.');
    }
    return chatId;
  }

  function _pickCreateMeta_(payload) {
    var meta = {};
    CREATE_META_FIELDS.forEach(function (field) {
      if (payload.hasOwnProperty(field) && payload[field] !== undefined) {
        meta[field] = payload[field];
      }
    });
    return meta;
  }

  // ============ create() → CreateTask ============

  /**
   * @param {object} payload  { title (必填), due_date?, priority?, category?,
   *                            recurring?, context?, budget?, notes?,
   *                            description?, tags? }
   * @param {object} context  { chatId (必填) }
   * @returns {object} ConnectorResponse；data 是创建（或幂等命中已存在）
   *                   的 task 对象
   */
  function create(payload, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'create', action: 'CreateTask', chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        var chatId = _requireChatId_(context);

        if (!ConnectorValidation.isPlainObject(payload) || !ConnectorValidation.isNonEmptyString(payload.title)) {
          throw ConnectorErrors.validation('TITLE_REQUIRED', 'payload.title (non-empty string) is required to create a task.');
        }

        var meta = _pickCreateMeta_(payload);
        var task = ProductivityOS.createTask(payload.title.trim(), meta, chatId);

        return {
          data: task,
          message: 'Task created. (Productivity OS creation is idempotent — resubmitting identical title/due_date/priority/category/recurring returns the pre-existing task instead of a duplicate; see Productivity OS 09_IdempotencyManager.gs.)'
        };
      }
    );
  }

  // ============ update() → UpdateTask ============

  /**
   * @param {string} id       task_id
   * @param {object} changes  可编辑字段的子集：title/category/priority/
   *                          due_date/recurring/context/budget/notes/
   *                          description/tags（哪些字段真正生效由
   *                          Productivity OS 自己的白名单决定，见 Notes）
   * @param {object} context  { chatId (必填) }
   * @returns {object} ConnectorResponse；data 是实际写入的字段（含可能
   *                   重算的 identity）
   */
  function update(id, changes, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'update', action: 'UpdateTask', chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        var chatId = _requireChatId_(context);

        if (!ConnectorValidation.isNonEmptyString(id)) {
          throw ConnectorErrors.validation('TASK_ID_REQUIRED', 'id (task_id, non-empty string) is required.');
        }
        if (!ConnectorValidation.isNonEmptyPlainObject(changes)) {
          throw ConnectorErrors.validation('CHANGES_REQUIRED', 'changes must be a non-empty object of fields to update.');
        }

        // 存在性预检查：updateTask() 对"任务不存在"和"传的字段都不在白
        // 名单内"这两种情况都返回 null，无法区分。这里先单独查一次存在性，
        // 只是为了把这两种失败原因分开报得更准确，不是重新实现更新逻辑
        // 本身（真正的更新仍然完全由 ProductivityOS.updateTask 执行）。
        var existing = ProductivityOS.TaskQueryEngine.getTask(id, chatId);
        if (!existing) {
          throw ConnectorErrors.business('TASK_NOT_FOUND', 'No task found with id "' + id + '" for this chatId.');
        }

        var updated = ProductivityOS.updateTask(id, changes, chatId);
        if (!updated) {
          throw ConnectorErrors.validation(
            'NO_UPDATABLE_FIELDS',
            'None of the supplied fields are updatable. Allowed fields: title, category, priority, due_date, recurring, context, budget, notes, description, tags.'
          );
        }

        return { data: updated, message: 'Task updated.' };
      }
    );
  }

  // ============ delete() → DeleteTask（映射为 cancelTask，见 Notes） ============

  /**
   * ⚠️ 语义映射说明（不是遗漏，是刻意选择）：
   * Productivity OS 是事件溯源架构，真相来源是不可变、只追加的 Events 表
   * （见其 00_Project_Constitution.gs），从设计上就没有"硬删除"这个操作——
   * 只有 create/update/complete/cancel 四种生命周期事件。因此 DeleteTask
   * 在这里实现为 cancelTask()（状态改为 CANCELLED 的软删除），而不是
   * 真的抹掉这条任务的存在痕迹。如果需要的是"完成"而不是"取消"，请用
   * execute('CompleteTask', ...)。
   *
   * @param {string} id       task_id
   * @param {object} context  { chatId (必填) }
   * @returns {object} ConnectorResponse
   */
  function del(id, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'delete', action: 'DeleteTask', chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        var chatId = _requireChatId_(context);

        if (!ConnectorValidation.isNonEmptyString(id)) {
          throw ConnectorErrors.validation('TASK_ID_REQUIRED', 'id (task_id, non-empty string) is required.');
        }

        var result = ProductivityOS.cancelTask(id, chatId);
        if (result && result.not_found) {
          throw ConnectorErrors.business('TASK_NOT_FOUND', 'No task found with id "' + id + '" for this chatId.');
        }

        return {
          data: { task_id: id, status: 'CANCELLED' },
          message: 'Task cancelled (soft delete — Productivity OS is event-sourced and has no hard-delete operation; see file header Notes).'
        };
      }
    );
  }

  // ============ get() → GetTask ============

  /**
   * @param {string} id       task_id
   * @param {object} [context] { chatId? }（传入时会额外校验 chat_id 匹配，
   *                             跨用户隔离，行为跟 TaskQueryEngine.getTask
   *                             一致）
   * @returns {object} ConnectorResponse；data 是单个 task 对象
   */
  function get(id, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'get', action: 'GetTask', chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        if (!ConnectorValidation.isNonEmptyString(id)) {
          throw ConnectorErrors.validation('TASK_ID_REQUIRED', 'id (task_id, non-empty string) is required.');
        }

        var task = ProductivityOS.TaskQueryEngine.getTask(id, context && context.chatId);
        if (!task) {
          throw ConnectorErrors.business('TASK_NOT_FOUND', 'No task found with id "' + id + '".');
        }

        return { data: task };
      }
    );
  }

  // ============ list() → ListTasks ============

  /**
   * @param {object} [filters]  简单等值过滤，比如 { status: 'PENDING' }
   *                             （复杂查询请用 search()）
   * @param {object} [context]  { chatId? }（不传则跨所有用户）
   * @returns {object} ConnectorResponse；data 是 task[] 数组
   */
  function list(filters, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'list', action: 'ListTasks', chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        if (filters !== undefined && filters !== null && !ConnectorValidation.isPlainObject(filters)) {
          throw ConnectorErrors.validation('INVALID_FILTERS', 'filters must be a plain object of { field: value } pairs, if provided.');
        }

        var tasks = ProductivityOS.TaskQueryEngine.getTasks(context && context.chatId, filters || undefined);
        return { data: tasks, metaExtra: { count: tasks.length } };
      }
    );
  }

  // ============ search() → SearchTasks ============

  /**
   * @param {string|object} query  字符串等价于 {text: query}；对象形式见
   *                                Productivity OS 23_SearchEngine.gs：
   *                                {text?, title?, category?, priority?,
   *                                 status?, recurring?, tag?, dateFrom?,
   *                                 dateTo?}
   * @param {object} [context]     { chatId? }
   * @returns {object} ConnectorResponse；data 是 task[] 数组
   */
  function search(query, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'search', action: 'SearchTasks', chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        if (!ConnectorValidation.isStringOrPlainObject(query)) {
          throw ConnectorErrors.validation('INVALID_QUERY', 'query must be a string or a plain object.');
        }

        var tasks = ProductivityOS.TaskQueryEngine.searchTasks(context && context.chatId, query);
        return { data: tasks, metaExtra: { count: tasks.length } };
      }
    );
  }

  // ============ execute() —— 具名 action 的通用分发入口 ============

  /**
   * 承载 create/update/delete/get/list/search 五个通用方法之外的具名动作。
   * 需求方要求的 CompleteTask/ReopenTask/TodayTasks/PriorityTasks/
   * Statistics 五个都在这里；额外多暴露了 Productivity OS
   * TaskQueryEngine 已有、但需求方名单没点名的几个只读视图（Week/Month/
   * Tomorrow/Overdue/Upcoming/Recurring/Cancelled/Archived/Dashboard），
   * 因为它们零额外实现成本（Productivity OS 已经有），符合"Future
   * Expansion...不需要为了加能力改架构"的要求——加的是"这个 Connector
   * 暴露的 action 清单"，不是新架构。
   *
   * @param {string} action  见 capabilities().actions 的完整清单
   * @param {object} [params]
   * @param {object} [context]  { chatId? }（CompleteTask 必填 chatId；
   *                             ReopenTask 会先校验 chatId 再判定不支持，
   *                             保持"先验证输入，再判断能不能做"的统一顺序）
   * @returns {object} ConnectorResponse
   */
  function execute(action, params, context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'execute', action: action, chatId: context && context.chatId },
      function () {
        _ensureLibraryLoaded_();
        params = params || {};

        if (!ConnectorValidation.isNonEmptyString(action)) {
          throw ConnectorErrors.validation('ACTION_REQUIRED', 'action (non-empty string) is required.');
        }

        switch (action) {

          case 'CompleteTask': {
            var chatIdComplete = _requireChatId_(context);
            if (!ConnectorValidation.isNonEmptyString(params.id)) {
              throw ConnectorErrors.validation('TASK_ID_REQUIRED', 'params.id (task_id) is required for CompleteTask.');
            }
            var completeResult = ProductivityOS.completeTask(params.id, chatIdComplete);
            if (completeResult && completeResult.not_found) {
              throw ConnectorErrors.business('TASK_NOT_FOUND', 'No task found with id "' + params.id + '" for this chatId.');
            }
            return {
              data: completeResult,
              message: completeResult && completeResult.next_task
                ? 'Task completed; a recurring next occurrence was created (see data.next_task).'
                : 'Task completed.'
            };
          }

          case 'ReopenTask': {
            // 先校验存在性/chatId，让"任务不存在"这种更具体的错误优先于
            // "这个能力目前不支持"的笼统提示。
            if (!ConnectorValidation.isNonEmptyString(params.id)) {
              throw ConnectorErrors.validation('TASK_ID_REQUIRED', 'params.id (task_id) is required for ReopenTask.');
            }
            var existingForReopen = ProductivityOS.TaskQueryEngine.getTask(params.id, context && context.chatId);
            if (!existingForReopen) {
              throw ConnectorErrors.business('TASK_NOT_FOUND', 'No task found with id "' + params.id + '".');
            }
            throw ConnectorErrors.business(
              'REOPEN_NOT_SUPPORTED',
              'Productivity OS has no reopen capability: TaskEngine.updateTask() deliberately excludes "status" from its UPDATABLE_FIELDS whitelist, and there is no TASK_REOPENED event type. Create a new task instead if this is genuinely needed again.',
              { taskId: params.id, currentStatus: existingForReopen.status }
            );
          }

          case 'TodayTasks':
            return { data: ProductivityOS.TaskQueryEngine.getTodayTasks(context && context.chatId) };

          case 'TomorrowTasks':
            return { data: ProductivityOS.TaskQueryEngine.getTomorrowTasks(context && context.chatId) };

          case 'WeekTasks':
            return { data: ProductivityOS.TaskQueryEngine.getWeekTasks(context && context.chatId) };

          case 'MonthTasks':
            return { data: ProductivityOS.TaskQueryEngine.getMonthTasks(context && context.chatId) };

          case 'OverdueTasks':
            return { data: ProductivityOS.TaskQueryEngine.getOverdueTasks(context && context.chatId) };

          case 'UpcomingTasks':
            return { data: ProductivityOS.TaskQueryEngine.getUpcomingTasks(context && context.chatId) };

          case 'RecurringTasks':
            return { data: ProductivityOS.TaskQueryEngine.getRecurringTasks(context && context.chatId) };

          case 'CancelledTasks':
            return { data: ProductivityOS.TaskQueryEngine.getCancelledTasks(context && context.chatId) };

          case 'ArchivedTasks':
            return { data: ProductivityOS.TaskQueryEngine.getArchivedTasks(context && context.chatId, params.limit) };

          case 'PriorityTasks':
            return { data: ProductivityOS.TaskQueryEngine.getPriorityTasks(context && context.chatId) };

          case 'Dashboard': {
            var dashboardType = ConnectorValidation.isNonEmptyString(params.type) ? params.type : 'today';
            return { data: ProductivityOS.TaskQueryEngine.getDashboard(dashboardType, context && context.chatId) };
          }

          case 'Statistics':
            return { data: ProductivityOS.TaskQueryEngine.getStatistics(context && context.chatId) };

          default:
            throw ConnectorErrors.validation(
              'UNKNOWN_ACTION',
              'Unknown action "' + action + '". See capabilities() for the full supported action list.',
              { action: action }
            );
        }
      }
    );
  }

  // ============ health() ============

  function health(context) {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'health', chatId: context && context.chatId },
      function () {
        if (typeof ProductivityOS === 'undefined') {
          return { data: { status: 'DOWN', reason: 'ProductivityOS library is not loaded in this Apps Script project.' } };
        }
        try {
          var stats = ProductivityOS.TaskQueryEngine.getStatistics(context && context.chatId);
          return {
            data: {
              status: 'UP',
              integrationMode: 'library',
              totalTasks: stats.total_count
            }
          };
        } catch (e) {
          return { data: { status: 'DEGRADED', reason: 'ProductivityOS library is loaded but a diagnostic call failed: ' + e.message } };
        }
      }
    );
  }

  // ============ capabilities() ============

  function capabilities() {
    return ConnectorSupport.handle(
      { connector: CONNECTOR_NAME, method: 'capabilities' },
      function () {
        return {
          data: {
            connector: CONNECTOR_NAME,
            connectorVersion: '1.0.0',
            domainOS: 'Productivity OS v4.3',
            integrationMode: 'library (Identifier: "ProductivityOS")',
            standardMethods: ['create', 'update', 'delete', 'get', 'list', 'search', 'execute', 'health', 'capabilities'],
            actions: {
              CreateTask:      { supported: true, via: 'create' },
              UpdateTask:      { supported: true, via: 'update' },
              DeleteTask:      { supported: true, via: 'delete', note: 'Maps to cancelTask (soft delete). Productivity OS is event-sourced and has no hard-delete operation.' },
              CompleteTask:    { supported: true, via: 'execute' },
              ReopenTask:      { supported: false, via: 'execute', reason: 'status is not in TaskEngine.updateTask\'s UPDATABLE_FIELDS whitelist; no TASK_REOPENED event type exists.' },
              GetTask:         { supported: true, via: 'get' },
              ListTasks:       { supported: true, via: 'list' },
              SearchTasks:     { supported: true, via: 'search' },
              TodayTasks:      { supported: true, via: 'execute' },
              PriorityTasks:   { supported: true, via: 'execute' },
              Statistics:      { supported: true, via: 'execute' },
              // Bonus actions — already free on the Productivity OS side (12_TaskQueryEngine.gs)
              TomorrowTasks:   { supported: true, via: 'execute' },
              WeekTasks:       { supported: true, via: 'execute' },
              MonthTasks:      { supported: true, via: 'execute' },
              OverdueTasks:    { supported: true, via: 'execute' },
              UpcomingTasks:   { supported: true, via: 'execute' },
              RecurringTasks:  { supported: true, via: 'execute' },
              CancelledTasks:  { supported: true, via: 'execute' },
              ArchivedTasks:   { supported: true, via: 'execute' },
              Dashboard:       { supported: true, via: 'execute' }
            }
          }
        };
      }
    );
  }

  return {
    create:       create,
    update:       update,
    delete:       del,
    get:          get,
    list:         list,
    search:       search,
    execute:      execute,
    health:       health,
    capabilities: capabilities
  };
})();
